45  /**
46   * A realm which extracts and dumps Kerberos {@code AuthorizationData}, but delegates the actual
47   * work to {@link ActiveDirectoryRealm#getPrincipal(GSSName, GSSCredential)}.
48   * <p>
49   * This realm requires your JVM to provide an {@link ExtendedGSSContext} implementation. It will use
50   * {@link InquireType#KRB5_GET_AUTHZ_DATA} to extract {@code AuthorizationData} according to RFC 4120,
51   * section 5.2.6 from an established security context, dump to
52   * <code>{catalina.base}/work/KRB5_AUTHZ_DATA/{gssName}/{yyyyMMdd'T'HHmmss.SSS}{#n?}</code> and
53   * continue as described.
54   * <p>
55   * <strong>Note</strong>: Use this realm for testing/analysis purposes only along with the
56   * {@link Krb5AuthzDataDumpPrinter}.
57   */
58  public class Krb5AuthzDataDumpingActiveDirectoryRealm extends ActiveDirectoryRealm {
59  
60  	private static final DateTimeFormatter TS_FORMAT = DateTimeFormatter
61  			.ofPattern("yyyyMMdd'T'HHmmss.SSS").withZone(ZoneId.systemDefault());
62  
63  	protected Principal getPrincipal(GSSName gssName, GSSCredential gssCredential,
64  			GSSContext gssContext) {
65  		if (gssContext instanceof ExtendedGSSContext) {
66  			ExtendedGSSContext extGssContext = (ExtendedGSSContext) gssContext;
67  
68  			AuthorizationDataEntry[] adEntries = null;
69  			try {
70  				adEntries = (AuthorizationDataEntry[]) extGssContext
71  						.inquireSecContext(InquireType.KRB5_GET_AUTHZ_DATA);
72  
73  			} catch (GSSException e) {
74  				logger.warn(sm.getString("krb5AuthzDataRealmBase.inquireSecurityContextFailed"), e);
75  			}
76  
77  			if (adEntries == null) {
78  				if (logger.isDebugEnabled())
79  					logger.debug(sm.getString("krb5AuthzDataRealmBase.noDataProvided", gssName));
80  				return null;
81  			}
82  
83  			File catalinaBase = getServer().getCatalinaBase();
84  			Path workDir = catalinaBase.toPath().resolve("work");
85  			Instant id = Instant.now();
86  
87  			Path dumpDir = workDir.resolve("KRB5_AUTHZ_DATA").resolve(gssName.toString());
88  			try {
89  				Path dumpFile = createDumpFile(dumpDir, id);
90  				try (PrintWriter w = new PrintWriter(Files.newBufferedWriter(dumpFile))) {
91  					for (AuthorizationDataEntry adEntry : adEntries) {
92  						w.printf("%d %s%n", adEntry.getType(),
93  								Base64.getEncoder().encodeToString(adEntry.getData()));
94  					}
95  				}
96  			} catch (IOException e) {
97  				logger.warn(sm.getString(
98  						"krb5AuthzDataDumpingActiveDirectoryRealm.dumpingKrb5AuthzDataFailed",
99  						gssName), e);
100 			}
101 		} else {
102 			logger.error(sm.getString("krb5AuthzDataRealmBase.incompatibleSecurityContextType"));
103 		}
104 
105 		return getPrincipal(gssName, gssCredential);
106 	}
107 
108 	private Path createDumpFile(Path dumpDir, Instant id) throws IOException {
109 		Files.createDirectories(dumpDir);
110 		String formattedTimestamp = TS_FORMAT.format(id);
111 		Path dumpFile = dumpDir.resolve(formattedTimestamp);
112 		int i = 2;
113 		while (Files.exists(dumpFile)) {
114 			dumpFile = dumpDir.resolve(formattedTimestamp + "#" + i++);
115 		}
116 		if (FileSystems.getDefault().supportedFileAttributeViews().contains("posix")) {
117 			Set<PosixFilePermission> ownerWritable = PosixFilePermissions.fromString("rw-------");
118 			FileAttribute<?> permissions = PosixFilePermissions.asFileAttribute(ownerWritable);
119 			Files.createFile(dumpFile, permissions);
120 		}
121 		return dumpFile;
122 	}
123 
124 }
