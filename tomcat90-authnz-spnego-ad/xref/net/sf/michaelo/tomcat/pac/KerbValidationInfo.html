29  /**
30   * A class representing the <a href=
31   * "https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-pac/69e86ccc-85e3-41b9-b514-7d969cd0ed73">{@code KERB_VALIDATION_INFO}</a>
32   * structure from MS-PAC. This implementation only parses the members which are required for the
33   * purpose of this component, everything else is skipped.
34   */
35  public class KerbValidationInfo {
36  
37  	public final static long EXTRA_SIDS_USER_FLAG = 0x00000020L;
38  	public final static long RESOURCE_GROUP_IDS_USER_FLAG = 0x00000200L;
39  
40  	protected final Log logger = LogFactory.getLog(getClass());

66  	/**
67  	 * Parses a Kerberos validation info object from a byte array.
68  	 *
69  	 * @param infoBytes
70  	 *            Kerberos validation info structure encoded as bytes
71  	 * @throws NullPointerException
72  	 *             if {@code infoBytes} is null
73  	 * @throws IllegalArgumentException
74  	 *             if {@code infoBytes} is empty
75  	 * @throws IllegalArgumentException
76  	 *             if {@code GroupCount} is not equal to the actually marshaled group count
77  	 * @throws IllegalArgumentException
78  	 *             if {@code SidCount} is bigger than 0, but flag D is not set in {@code UserFlags}
79  	 * @throws IllegalArgumentException
80  	 *             if {@code SidCount} is not equal to the actually marshaled SID count
81  	 * @throws IllegalArgumentException
82  	 *             if {@code ResourceGroupDomainSid} is not {@code null}, but flag H is not set in
83  	 *             {@code UserFlags}
84  	 * @throws IllegalArgumentException
85  	 *             if {@code ResourceGroupCount} is bigger than 0, but flag H is not set in
86  	 *             {@code UserFlags}
87  	 * @throws IllegalArgumentException
88  	 *             if {@code ResourceGroupCount} is not equal to the actually marshaled resource
89  	 *             group count
90  	 * @throws IllegalArgumentException
91  	 *             if any {@code RPC_UNICODE_STRING} is incorrectly NDR-encoded
92  	 */
93  	public KerbValidationInfo(byte[] infoBytes) {
94  		Objects.requireNonNull(infoBytes, "infoBytes cannot be null");
95  		if (infoBytes.length == 0)
96  			throw new IllegalArgumentException("infoBytes cannot be empty");
97  
98  		PacDataBuffer buf = new PacDataBuffer(infoBytes);
99  
100 		// common RPC header
101 		buf.skip(8);
102 		// RPC type marshalling private header
103 		buf.skip(8);
104 		// RPC unique pointer
105 		long uniquePointer = buf.getUnsignedInt();
106 		logPointer("RPC unique", uniquePointer);
107 
108 		// LogonTime
109 		buf.skip(8);
110 		// LogoffTime
111 		buf.skip(8);
112 		// KickOffTime
113 		buf.skip(8);
114 		// PasswordLastSet
115 		buf.skip(8);
116 		// PasswordCanChange
117 		buf.skip(8);
118 		// PasswordMustChange
119 		buf.skip(8);
120 		// EffectiveName
121 		RpcUnicodeString effectiveName = getRpcUnicodeString(buf);
122 		// FullName
123 		RpcUnicodeString fullName = getRpcUnicodeString(buf);
124 		// LogonScript
125 		RpcUnicodeString logonScript = getRpcUnicodeString(buf);
126 		// ProfilePath
127 		RpcUnicodeString profilePath = getRpcUnicodeString(buf);
128 		// HomeDirectory
129 		RpcUnicodeString homeDirectory = getRpcUnicodeString(buf);
130 		// HomeDirectoryDrive
131 		RpcUnicodeString homeDirectoryDrive = getRpcUnicodeString(buf);
132 		// LogonCount
133 		buf.skip(2);
134 		// BadPasswordCount
135 		buf.skip(2);
136 		// UserId
137 		this.userId = buf.getUnsignedInt();
138 		// PrimaryGroupId
139 		this.primaryGroupId = buf.getUnsignedInt();
140 		// GroupCount
141 		long groupCount = buf.getUnsignedInt();
142 		// GroupIds
143 		long groupIdsPointer = buf.getUnsignedInt();
144 		// UserFlags
145 		/* Something isn't right, it appears to be that the bits are in reverse order
146 		 * or the documentation is wrong:
147 		 * - flag H should be at bit 22, but is at bit 9
148 		 * - flag D should be at bit 26, but is at bit 5
149 		 *
150 		 * Samba has the same reversed order: https://github.com/samba-team/samba/blob/9844ac289be3430fd3f72c5e57fa00e012c5d417/librpc/idl/netlogon.idl#L251-L263
151 		 */
152 		this.userFlags = buf.getUnsignedInt();
153 		// UserSessionKey
154 		buf.skip(16);
155 		// LogonServer
156 		RpcUnicodeString logonServer = getRpcUnicodeString(buf);
157 		// LogonDomainName
158 		RpcUnicodeString logonDomainName = getRpcUnicodeString(buf);
159 		// LogonDomainId
160 		long logonDomainIdPointer = buf.getUnsignedInt();
161 		// Reserved1
162 		buf.skip(8);
163 		// UserAccountControl
164 		/*
165 		 * This is NOT userAccountControl from LDAP, see
166 		 * https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-samr/b10cfda1-f24f-441b-
167 		 * 8f43-80cb93e786ec
168 		 */
169 		this.userAccountControl = buf.getUnsignedInt();
170 		// SubAuthStatus
171 		buf.skip(4);
172 		// LastSuccessfulILogon
173 		buf.skip(8);
174 		// LastFailedILogon
175 		buf.skip(8);
176 		// FailedILogonCount
177 		buf.skip(4);
178 		// Reserved3
179 		buf.skip(4);
180 		// SidCount
181 		long sidCount = buf.getUnsignedInt();
182 		// ExtraSids
183 		long extraSidsPointer = buf.getUnsignedInt();
184 		// ResourceGroupDomainSid
185 		long resourceGroupDomainSidPointer = buf.getUnsignedInt();
186 		// ResourceGroupCount
187 		long resourceGroupCount = buf.getUnsignedInt();
188 		// ResourceGroupIds
189 		long resourceGroupIdsPointer = buf.getUnsignedInt();
190 
191 		logPointer("effectiveName", effectiveName.getPointer());
192 		this.effectiveName = getNdrString(buf, effectiveName);
193 		logPointer("fullName", fullName.getPointer());
194 		this.fullName = getNdrString(buf, fullName);
195 		logPointer("logonScript", logonScript.getPointer());
196 		this.logonScript = getNdrString(buf, logonScript);
197 		logPointer("profilePath", profilePath.getPointer());
198 		this.profilePath = getNdrString(buf, profilePath);
199 		logPointer("homeDirectory", homeDirectory.getPointer());
200 		this.homeDirectory = getNdrString(buf, homeDirectory);
201 		logPointer("homeDirectoryDrive", homeDirectoryDrive.getPointer());
202 		this.homeDirectoryDrive = getNdrString(buf, homeDirectoryDrive);
203 
204 		long actualGroupCount = buf.getUnsignedInt();
205 		if (groupCount != actualGroupCount)
206 			throw new IllegalArgumentException("GroupCount is " + groupCount
207 					+ ", but actual GroupCount is " + actualGroupCount);
208 
209 		logPointer("groupIds", groupIdsPointer);
210 		groupIds = new ArrayList<GroupMembership>();
211 		for (long l = 0L; l < groupCount; l++) {
212 			long relativeId = buf.getUnsignedInt();
213 			long attributes = buf.getUnsignedInt();
214 			groupIds.add(new GroupMembership(relativeId, attributes));
215 		}
216 
217 		logPointer("logonServer", logonServer.getPointer());
218 		this.logonServer = getNdrString(buf, logonServer);
219 		logPointer("logonDomainName", logonDomainName.getPointer());
220 		this.logonDomainName = getNdrString(buf, logonDomainName);
221 		logPointer("logonDomainId", logonDomainIdPointer);
222 		logonDomainId = getRpcSid(buf);
223 
224 		if (sidCount != 0L && (userFlags & EXTRA_SIDS_USER_FLAG) == 0L)
225 			throw new IllegalArgumentException("SidCount is " + sidCount
226 					+ ", but flag D is not set in UserFlags (" + toHexString(userFlags) + ")");
227 
228 		if ((userFlags & EXTRA_SIDS_USER_FLAG) != 0L) {
229 			logPointer("extraSids", extraSidsPointer);
230 			extraSids = new ArrayList<>();
231 			long actualSidCount = buf.getUnsignedInt();
232 			if (sidCount != actualSidCount)
233 				throw new IllegalArgumentException(
234 						"SidCount is " + sidCount + ", but actual SidCount is " + actualSidCount);
235 			long[] sidAttrs = new long[(int) sidCount];
236 			for (long l = 0L; l < sidCount; l++) {
237 				long extraSidPointer = buf.getUnsignedInt();
238 				long attributes = buf.getUnsignedInt();
239 				sidAttrs[(int) l] = attributes;
240 				logPointer("extraSid (" + l + ")", extraSidPointer);
241 			}
242 			for (long l = 0L; l < sidCount; l++) {
243 				Sid extraSid = getRpcSid(buf);
244 				extraSids.add(new KerbSidAndAttributes(extraSid, sidAttrs[(int) l]));
245 			}
246 		}
247 
248 		if (resourceGroupDomainSidPointer != 0L && (userFlags & RESOURCE_GROUP_IDS_USER_FLAG) == 0L)
249 			throw new IllegalArgumentException("ResourceGroupDomainSid is not null ("
250 					+ toHexString(resourceGroupDomainSidPointer)
251 					+ "), but flag H is not set in UserFlags (" + toHexString(userFlags) + ")");
252 
253 		if (resourceGroupCount != 0L && (userFlags & RESOURCE_GROUP_IDS_USER_FLAG) == 0L)
254 			throw new IllegalArgumentException("ResourceGroupCount is " + sidCount
255 					+ ", but flag H is not set in UserFlags (" + toHexString(userFlags) + ")");
256 
257 		if ((userFlags & RESOURCE_GROUP_IDS_USER_FLAG) != 0L) {
258 			logPointer("resourceGroupDomainSid", resourceGroupDomainSidPointer);
259 			resourceGroupDomainSid = getRpcSid(buf);
260 			long actualResourceGroupCount = buf.getUnsignedInt();
261 			if (resourceGroupCount != actualResourceGroupCount)
262 				throw new IllegalArgumentException("ResourceGroupCount is " + resourceGroupCount
263 						+ ", but actual ResourceGroupCount is " + actualResourceGroupCount);
264 
265 			logPointer("resourceGroupIds", resourceGroupIdsPointer);
266 			resourceGroupIds = new ArrayList<>();
267 			for (long l = 0L; l < resourceGroupCount; l++) {
268 				long relativeId = buf.getUnsignedInt();
269 				long attributes = buf.getUnsignedInt();
270 				resourceGroupIds.add(new GroupMembership(relativeId, attributes));
271 			}
272 		}
273 	}

344 	private RpcUnicodeString getRpcUnicodeString(PacDataBuffer buf) {
345 		int length = buf.getUnsignedShort();
346 		int maximumLength = buf.getUnsignedShort();
347 		if (maximumLength % 2 == 1)
348 			maximumLength -= 1;
349 		long pointer = buf.getUnsignedInt();
350 
351 		return new RpcUnicodeString(length, maximumLength, pointer);
352 	}
353 
354 	/* See:
355 	 *  - https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-rpce/ad703c18-564d-4238-a371-8d43cf442f81
356 	 *  - https://pubs.opengroup.org/onlinepubs/9629399/chap14.htm#tagcjh_19_03_04_02
357 	 */
358 	private String getNdrString(PacDataBuffer buf, RpcUnicodeString string) {
359 		long maximumCount = buf.getUnsignedInt();
360 		long offset = buf.getUnsignedInt();
361 		long actualCount = buf.getUnsignedInt();
362 
363 		if (offset > maximumCount || actualCount > maximumCount - offset)
364 			throw new IllegalArgumentException(
365 					"Incorrectly NDR-encoded UNICODE_STRING: maximumCount: " + maximumCount
366 							+ ", offset: " + offset + ", actualCount: " + actualCount);
367 
368 		if (maximumCount != string.getMaximumLength() / 2L
369 				|| actualCount != string.getLength() / 2L)
370 			throw new IllegalArgumentException(
371 					"NDR-encoded UNICODE_STRING does not match RPC_UNICODE_STRING: maximumCount: "
372 							+ maximumCount + ", actualCount: " + actualCount + ", maximumLength: "
373 							+ string.getMaximumLength() + ", length: " + string.getLength());
374 
375 		buf.skip(2 * (int) offset);
376 
377 		byte[] dst = new byte[2 * (int) actualCount];
378 		buf.get(dst);
379 
380 		return new String(dst, StandardCharsets.UTF_16LE);
381 	}
382 
383 	private Sid getRpcSid(PacDataBuffer buf) {
384 		long actualSubAuthorityCount = buf.getUnsignedInt();
385 		byte[] sidBytes = new byte[8 + (int) actualSubAuthorityCount * 4];
386 		buf.get(sidBytes);
387 		return new Sid(sidBytes);
388 	}
